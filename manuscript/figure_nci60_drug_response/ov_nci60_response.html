FDA Approved and Clinical Trial Compounds

Get Data

# Example filter on particular properties of the compounds
nscs <- sqldf("SELECT nsc
                FROM df 
                WHERE (testing_status != 'NA' AND testing_status != '')")
## Loading required package: tcltk
nscs <- nscs$nsc
length(nscs)
## [1] 310
# Get indicies for selected NSCs
idx <- which(rownames(drugDB$act) %in% nscs)

# Generate new data frame
data <- c(drugDB$act[idx,"OV.OVCAR_3"], drugDB$act[idx,"OV.OVCAR_4"], drugDB$act[idx,"OV.IGROV1"])
labels <- c(rep("OV.OVCAR_3", length(idx)), rep("OV.OVCAR_4", length(idx)), rep("OV.IGROV1", length(idx)))

tmp = data.frame(data=data, labels=labels)

Calculate the K-S Test

splitData <- split(tmp, tmp$labels)

p <- ggplot(tmp, aes(x=data)) + stat_ecdf(aes(colour=labels))
p <- p + ggtitle("Empirical Cumulative Distribution Function FDA Approved \n and Clinical Trial Drugs in the NCI-60")
p <- p + xlab("Z-Score") + ylab("Fraction")
p <- p + labs(colour = "Cell Lines")
plot(p)

ks.test(splitData[["OV.IGROV1"]]$data, splitData[["OV.OVCAR_3"]]$data, alternative="two.sided")
## 
##  Two-sample Kolmogorov-Smirnov test
## 
## data:  splitData[["OV.IGROV1"]]$data and splitData[["OV.OVCAR_3"]]$data
## D = 0.1045, p-value = 0.07202
## alternative hypothesis: two-sided
ks.test(splitData[["OV.IGROV1"]]$data, splitData[["OV.OVCAR_4"]]$data, alternative="two.sided")
## 
##  Two-sample Kolmogorov-Smirnov test
## 
## data:  splitData[["OV.IGROV1"]]$data and splitData[["OV.OVCAR_4"]]$data
## D = 0.2236, p-value = 5.274e-07
## alternative hypothesis: two-sided
